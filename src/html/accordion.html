Events

시점에 따른 이벤트

document.querySelectorAll('[data-accr-item]').forEach(function (el) {
    // 열릴 때
    el.addEventListener('modal.showing', function () {
        console.log(this, 'showing');
    })
    // 열린 후
    el.addEventListener('modal.shown', function () {
        console.log(this, 'shown');
    })
    // 닫힐 때
    el.addEventListener('modal.hiding', function () {
        console.log(this, 'hiding');
    })
    // 닫힌 후
    el.addEventListener('modal.hidden', function () {
        console.log(this, 'hidden');
    })
})

Method

Component 호출

const accrEl = document.querySelector('#accordion1');
const accr = new SKY.Accordion(accrEl);

전체 열기 ([data-accr="only"]에선 안됩니다.)

accr.showAll();

전체 닫기

accr.showAll();

특정 아코디언 열기

accr.show(2); // number
accr.show('#accordion1_3'); // string
accr.show(document.querySelector('#accordion1_3')); // object

특정 아코디언 닫기

accr.show(2); // number
accr.hide('#accordion1_3'); // string
accr.hide(document.querySelector('#accordion1_3')); // object
